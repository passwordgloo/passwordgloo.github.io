GTK_IM_MODULE=fcitx
QT_IM_MODULE=fcitx
XMODIFIERS=@im=fcitx
INPUT_METHOD=fcitx
SDL_IM_MODULE=fcitx
GLFW_IM_MODULE=ibus

                                  # Youtube-dl(國外更佳)

                                  # 安裝

                                  下載位置為當前終端目錄

                                  python3 --version								//查看Python版本
sudo apt install python3-pip					//安裝pip，如果有請忽略
sudo -H pip install --upgrade youtube-dl		//安裝YouTube-dl

                                  # 格式

                                  youtube-dl XXXXXXXXX        //下載
--write-sub			//下載視頻自帶字幕
--sub-lang en		//下載指定語言字幕
--write-auto-sub		//視頻無字幕，使用AI字幕
-f "bestvideo+bestaudio"		//下載最佳音頻和視頻文件
-o '%(title)s.%(ext)s'   //指定文件名與文件格式
-a '.txt'        //從文本提取鏈接下載

                                  # 技巧

                                  1. 如果ubnutu自帶vscode不能輸入中文，請上官網下載
                                  2. Vscode保存文件提示權限不夠
                                  3. Vscode調試c/c++出現文件不存在報錯,但正常編譯，請檢查文件命名開頭有無中文
                                  4. 出現找不到targetArchitectureq請在laugh.json添加 "targetArchitecture": "x86_64"
                                  5. 出現中文亂碼，請在vs code中修改編碼方式為GBK保存
                                  6. 調用cmd測試壹閃而過，請在c/c++文件中插入以下字符：
                                  #include　<stdlib.h>
system("pause");       //放在return語句前面

                                  # 手动下载

                                  1. 訪問docsetsopen in new window
                                  2. ctrl + F搜索文檔名，記錄name值
                                  3. 下載格式：http://{city_name}.kapeli.com/feeds/{name}.tgz
                                    • {city_name}：frankfurt；london；newyork；sanfrancisco；singapore；sydney；tokyo
                                    • {name}：name
