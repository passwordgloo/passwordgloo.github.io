                    # 安裝

                    # 更新Ubnutu服務器

                    sudo apt update
sudo apt upgrade

                    # 安裝Apache Web服務器

                    sudo apt install apache2

                    # 檢查Apache Web狀態

                    sudo systemctl status apache2

                    # 配置

                    # Apache環境設置open in new window

                    # 防火墻設置[可選]

                    # 查看可用應用程序配置文件

                    sudo ufw app list

                    # 列出配置文件及包含规则信息

                    sudo ufw app info "Apache Full"

                    # 允許HTTP/HTTPS傳入通信

                    sudo ufw allow in "Apache Full"

                    # 輸入localhost訪問測試頁面

                    # Mysql

                    # 安裝

                    # 安裝命令

                    sudo apt install mysql-server

                    # 檢查Mysql服務是否運行

                    sudo systemctl status mysql

                    # Phpmyadmin

                    sudo apt install phpmyadmin
sudo ln -s /usr/share/phpmyadmin /var/www/phpmyadmin   //软链接

                    # PHP

                    # 安裝

                    # 安裝命令

                    sudo apt install php libapache2-mod-php php-mysql
sudo apt-get install php8.1-fpm                      //可選

                    # 測試

                    sudo vi /var/www/html/index.php    //新建index.php

                    //index.php輸入以下代碼，按:wq保存退出
<? php                            
phpinfo();
?>

                    sudo systemctl restart apache2    //重啟Apache服務

                    # Host

                    vim /etc/hosts
