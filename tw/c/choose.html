                                                        # 24.判斷兩個實型數據是否相等

                                                        #include <stdio.h>
#include <math.h>
int main(void)
{
	float a = -2.12,b = -10.33;
	if(fabs(a-b) <= 1e-4)                           //double fabs(double), int abs(int)
		printf("這兩個數相等");
	else
		printf("這兩個數不等");
	return 0;
}

                                                        # 25.求壹個壹元二次方程的根

                                                        #include <stdio.h>
#include <math.h>
int main(void)
{
	double a,b,c,d;
	printf("請輸入壹元二次方程的3個系數");
	scanf("%lf%lf%lf",&a,&b,&c,&d);
	if(fabs(a=0) <=1e-6)
	{
		printf("輸入有誤,程序結束");
		return 0;
	}
	d = b*b-4*a*c;
	if(d<0)
		printf("此方程無實數根");
	else  if(fabs(d) <=1e-6)
		printf("此方程有兩個相同的實數根:%lf\n",-b/(2*a));
	else
		printf("此方程的兩個實數根為:%lf,&lf\n",(-b+sqrt(d))/2*a,(-b-sqrt(d))/2*a);
	return 0;
}

                                                        # 26.邏輯型數據

                                                        #include <stdbool.h>                //該文件中有_Bool這個數據類型，_Bool與bool是同義詞，vc++沒有這個文件
#include <stdio.h>
int main(void)
{
	//定義布爾型變量——c99新增變量
	bool a = false;                    //等價於bool a = 0;
	bool b = true;                     //等價於bool b = 1;
//	_bool c;                           //與bool c;等價


	b = 3;
	printf("%d\n",b);
	printf("%\n",sizeof(a));           //布爾型數據占1個字節
	return 0;

}

                                                        # 27.switch語句

                                                        #include <stdio.h>
int main(void)
{
	printf("請輸入所要到達的樓層:");
	int a;
	scanf("%d",&a);
	switch(a)
	{
	default:printf("沒有這個樓層\n");break;
		case 1:
			printf("上壹樓\n");
			break;
		case 2:
			printf("上二樓\n");
			break;
		case 3:
			printf("上三樓\n");
			break;
	}
	return 0;
}

                                                        # 29.頭文件

                                                        #include <文件名> 系統到存放C庫函數頭文件的目錄中尋找要包含的文件，這稱為標準方式。 #include "文件名"系統現在用戶當前目錄中尋找要包含的文件，要是找不到，再按標準方式查找。

                                                        頭文件的後綴可以是'.c'、'.h'或者沒有

                                                        #include <stdio.h>                         //調用printf函數
#define P1 =3.14159

int E = 123;
int max(int x,int y)
{
	x>y?printf("%d",x):printf("%d",x=y);
}
