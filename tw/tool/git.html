                                        # 安裝

                                        sudo apt-get install git      		//普通用戶安裝
apt-get install git           		//root權限

                                        # 全局設置

                                        git config --global user.name "passwordgloo"    
git config --global user.email "passwordgloo@163.com"

                                        # 初始化

                                        mkdir repository_name               //新建倉庫文件夾
cd repository_name
git init  							//初始化，會生成git隱藏文件夾
git remote add origin git地址  	   //鏈接本地倉庫到遠程倉庫

                                        # 基本操作

                                        git add file  						//添加文件到暫存區
git add .  							//添加當前工作區所有修改的文件到暫存區
git reset HEAD 文件名  			  //撤銷暫存區操作
git checkout -- 文件名  			  //放棄工作區更改
git commit -m "commit about file"   //添加評論並提交版本庫
git push -u origin master           //推送到遠程倉庫

                                        # 回退版本

                                        git log  							//git日誌
git reflog  						//git簡略版本日誌
git show-ref						//git查看ref
git reset --hard 版本號  			  //切換版本，comit前十位可確定版本號
git push origin HEAD --force		//強制同步遠程倉庫

                                        # 分支管理

                                        使用新的git switch命令，比git checkout更容易理解

                                        # 創建分支

                                        # 寫法1

                                        git branch dev						//創建分支dev
git checkout dev					//切換到dev分支

                                        # 寫法2

                                        git checkout -b dev  				//創建並切換到dev分支

                                        # 寫法3

                                        git switch -c dev					//創建並切換到dev分支

                                        # 切換分支

                                        # 寫法1

                                        git checkout master 				//舊版本，切換到master分支

                                        # 寫法2

                                        git switch master					//新版本，切換到master分支

                                        # 合並分支

                                        git merge dev						//合並dev分支到當前分支
git merge --no-ff dev				//禁用fast forward，合並dev
git log --graph						//查看分支合並圖

                                        # 刪除分支

                                        git branch -d dev					//刪除dev分支
git branch -D dev					//強行刪除未合並過的dev分支

                                        # 分支快照

                                        # 創建快照

                                        git stash							//將當前分支現場儲存起來
git stash list						//查看快照信息

                                        # 恢復快照

                                        git stash pop						//恢復並刪除快照

                                        # 多人协作

                                        git push origin <branch-name> 		//推送自己修改的分支
git pull							//如果推送失敗，試圖合並沖突

                                        git pull提示no tracking information，說明本地分支與遠程分支未創建鏈接，輸入git branch --set-upstream-to <branch-name> origin/<branch-name>

                                        # SSH

                                        # 生成密鑰

                                        # 方法1

                                        ssh-keygen -t rsa -C "xxx@xxx"		//指定郵箱地址生成rsa密鑰並選擇保存位置

                                        # 方法2

                                        ssh-keygen -t rsa -C "passwordgloo@163.com" -f ~/.ssh/github_id_rsa	//指定郵箱地址生成rsa密鑰並復制到指定位置

                                        # 添加私钥

                                        cd ~/.ssh
ssh-add github_id_rsa
ssh-agent bash 					//出现Could not open a connection to your authentication agent输入

                                        # 多个ssh共存

                                        • 當前目錄新建config文件（沒有文件類型名），輸入
                                        # gitee
    Host gitee
    HostName gitee.com
    PreferredAuthentications publickey
    IdentityFile ~/.ssh/gitee_id_rsa
    user git

# github
    Host github
    HostName github.com
    PreferredAuthentications publickey
    IdentityFile ~/.ssh/github_id_rsa
    user git
                                        • 執行sudo chmod 0600 config

                                        # 添加公鑰

                                        cat ~/.ssh/gitee_id_rsa.pub

                                        # 測試

                                        ssh -T git@github.com
ssh -T git@gitee.com

                                        # GPG

                                        # 創建

                                        gpg --full-generate-key 					  //新建公私鑰

                                        # 添加公鑰

                                        gpg --list-key 				  //列出生成的公私鑰
gpg --armor --export 哈希值    //輸出公鑰

                                        # 配置

                                        git config --global user.signingkey 哈希值 	//git設置密鑰信息
git config --global commit.gpgsign true 	  //設置全局簽名驗證
git commit -s '' 							  //gpg驗證的commit
